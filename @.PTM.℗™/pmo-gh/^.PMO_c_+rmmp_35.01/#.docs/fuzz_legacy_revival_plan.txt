### Phase 2: Implement PDL Reading (Priority: MEDIUM)

**Goal:** Replace hardcoded handlers with PDL-based dispatch.

#### Task 2.1: Create PDL Reader Utility

**File:** `pieces/system/pdl/pdl_reader.c`

```c
/*
 * pdl_reader.c - Read PDL files and extract method handlers
 * CPU-SAFE: No dynamic allocation, fixed buffers
 */

#ifndef _GNU_SOURCE
#define _GNU_SOURCE
#endif

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

#define MAX_PATH 4096
#define MAX_LINE 1024

static char* trim_str(char *str) {
    char *end;
    if(!str) return str;
    while(isspace((unsigned char)*str)) str++;
    if(*str == 0) return str;
    end = str + strlen(str) - 1;
    while(end > str && isspace((unsigned char)*end)) end--;
    end[1] = '\0';
    return str;
}

/**
 * Get method handler for a piece's event
 * 
 * @param piece_id The piece name (e.g., "zombie_01")
 * @param event The event name (e.g., "on_turn")
 * @param out_handler Buffer to store handler path
 * @param out_size Size of out_handler buffer
 * @param project_root Root project path
 * @param current_project Current project name
 * @return 0 on success (found), -1 if not found
 */
int get_method(const char* piece_id, const char* event, 
               char* out_handler, size_t out_size,
               const char* project_root, const char* current_project) {
    
    char pdl_path[MAX_PATH];
    snprintf(pdl_path, sizeof(pdl_path), "%s/projects/%s/pieces/%s/piece.pdl",
             project_root, current_project, piece_id);
    
    FILE *f = fopen(pdl_path, "r");
    if (!f) return -1;
    
    char line[MAX_LINE];
    while (fgets(line, sizeof(line), f)) {
        /* Look for: METHOD | event | handler */
        if (strncmp(line, "METHOD", 6) == 0) {
            char *pipe1 = strchr(line, '|');
            if (!pipe1) continue;
            
            char *pipe2 = strchr(pipe1 + 1, '|');
            if (!pipe2) continue;
            
            /* Extract event name */
            char parsed_event[64];
            size_t event_len = pipe2 - pipe1 - 1;
            if (event_len >= sizeof(parsed_event)) event_len = sizeof(parsed_event) - 1;
            strncpy(parsed_event, pipe1 + 1, event_len);
            parsed_event[event_len] = '\0';
            trim_str(parsed_event);
            
            /* Extract handler path */
            char parsed_handler[MAX_PATH];
            strncpy(parsed_handler, pipe2 + 1, sizeof(parsed_handler) - 1);
            parsed_handler[sizeof(parsed_handler) - 1] = '\0';
            trim_str(parsed_handler);
            
            if (strcmp(parsed_event, event) == 0) {
                strncpy(out_handler, parsed_handler, out_size - 1);
                fclose(f);
                return 0;  /* Found! */
            }
        }
    }
    
    fclose(f);
    return -1;  /* Not found */
}
```

char handler[MAX_PATH];
char *default_trait = build_path_malloc("pieces/apps/playrm/ops/+x/move_entity.+x");

/* Try PDL first, fallback to default */
if (get_method(active_target_id, "on_move", handler, sizeof(handler),
               project_root, current_project) == 0) {
    /* Use PDL-defined handler */
    pid_t p = fork();
    if (p == 0) {
        freopen("/dev/null", "w", stdout);
        freopen("/dev/null", "w", stderr);
        execl(handler, handler, active_target_id, dir_str, NULL); exit(1);
    } else {
        waitpid(p, NULL, 0);
    }
} else {
    /* Fallback to default */
    pid_t p = fork();
    if (p == 0) {
        freopen("/dev/null", "w", stdout);
        freopen("/dev/null", "w", stderr);
        execl(default_trait, default_trait, active_target_id, dir_str, NULL); exit(1);
    } else {
        waitpid(p, NULL, 0);
    }
}
